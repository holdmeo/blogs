// ==UserScript==
// @name         YouTube 双字幕 - 实时稳定版
// @namespace    http://tampermonkey.net/
// @version      3.1
// @description  实时英文 + 句子级中文翻译（稳定版）
// @match        https://www.youtube.com/*
// @grant        GM_addStyle
// @grant        GM_xmlhttpRequest
// @connect      translate.googleapis.com
// @run-at       document-end
// ==/UserScript==

(function () {

    'use strict';

    let subtitleContainer;
    let lastEnglish='';
    let lastChinese='';

    let sentenceBuffer='';
    let lastCaption='';
    let pauseCaption='';
    let lastUpdate=0;

    const SENTENCE_TIMEOUT=700;

    let lastTranslateTime = 0;
    const TRANSLATE_INTERVAL = 1200; // 1.2秒

    GM_addStyle(`
        .ytp-caption-segment{
            font-size:0!important;
            color:transparent!important;
        }
    `);

    function createUI(){

        if(subtitleContainer) return;

        subtitleContainer=document.createElement('div');

        subtitleContainer.style.position='absolute';
        subtitleContainer.style.bottom='12%';
        subtitleContainer.style.left='40px';
        subtitleContainer.style.zIndex='9999';
        subtitleContainer.style.pointerEvents='none';

        const style=`
        font-size:26px;
        font-family:Arial;
        line-height:1.35;
        text-shadow:0 0 6px black;
        `;

        const en=document.createElement('div');
        en.id='dual-en';
        en.style=style+"color:white";

        const zh=document.createElement('div');
        zh.id='dual-zh';
        zh.style=style+"color:yellow";

        subtitleContainer.appendChild(en);
        subtitleContainer.appendChild(zh);

        const video=document.querySelector('.html5-video-container');

        if(video) video.parentElement.appendChild(subtitleContainer);

    }

    function render(en,zh){

        createUI();

        document.getElementById('dual-en').textContent=en;
        document.getElementById('dual-zh').textContent=zh;
    }

    function translate(text,callback){

        const url=`https://translate.googleapis.com/translate_a/single?client=gtx&sl=auto&tl=zh-CN&dt=t&q=${encodeURIComponent(text)}`;

        GM_xmlhttpRequest({
            method:'GET',
            url:url,
            onload(res){

                try{

                    const data=JSON.parse(res.responseText);
                    const zh=data[0].map(e=>e[0]).join('');

                    callback(zh);

                }catch{

                    callback('');

                }

            },
            onerror(){
                callback('');
            }
        });

    }

    function sentenceEnd(text){

        return /[.!?。！？]$/.test(text) || text.length>120;

    }

    function getCaption(){

        const seg=document.querySelectorAll('.ytp-caption-segment');

        if(!seg.length) return lastCaption;

        const text=Array.from(seg)
            .map(e=>e.textContent.trim())
            .join(' ')
            .replace(/\s+/g,' ')
            .trim();

        if(text) lastCaption=text;

        return text||lastCaption;

    }

    function isVideoPlaying(){
        const video=document.querySelector('video');
        return video && !video.paused && !video.ended;
    }

    function translateCurrentCaption(){
        const text = getCaption();
        if(!text) return;
        if(pauseCaption === text) return;
        translate(text, zh => {
            lastChinese = zh;
            pauseCaption=text;
            render(text, zh);
        });
    }

    function process(){

        if(!isVideoPlaying() && !document.hidden) return;

        const text=getCaption();

        if(!text) return;

        if(text===lastEnglish) return;

        lastEnglish=text;

        render(text,lastChinese);

        sentenceBuffer=text;

        sentenceBuffer=sentenceBuffer.trim();

        if(!sentenceBuffer) return;
        lastUpdate=Date.now();
        const now = Date.now();
        const timeout=(now-lastUpdate)>SENTENCE_TIMEOUT;
        const canTranslate = (now - lastTranslateTime) > TRANSLATE_INTERVAL;

        if(sentenceEnd(sentenceBuffer)|| canTranslate ||timeout){
            const finalText=sentenceBuffer;
            sentenceBuffer='';
            translate(finalText,zh=>{
                lastChinese=zh;
                lastTranslateTime = Date.now();
                render(lastEnglish,zh);
            });

        }
    }

    let observer=null;

    function observeCaptions(){
        const target=document.querySelector('.ytp-caption-window-container');
        if(!target) return;
        if(observer){
            observer.disconnect();
        }
        observer=new MutationObserver(process);
        observer.observe(target,{
            childList:true,
            subtree:true,
            characterData:true
        });

    }

    function reset(){
        console.log('字幕系统重置');
        lastEnglish='';
        lastChinese='';
        sentenceBuffer='';
        lastCaption='';

        if(subtitleContainer){
            subtitleContainer.remove();
            subtitleContainer=null;
        }
        waitVideo(); // 重新监听
    }

    function waitVideo(){
        let retry=0;
        const timer=setInterval(()=>{
            const cap=document.querySelector('.ytp-caption-window-container');
            const video=document.querySelector('video');
            if(cap && video){
                clearInterval(timer);
                observeCaptions();

                video.addEventListener('pause', () => {
                    observeCaptions();
                    setTimeout(translateCurrentCaption, 300);
                });

                video.addEventListener('seeked', () => {
                    observeCaptions();
                    setTimeout(translateCurrentCaption, 250);
                });
            }
            if(retry++ > 15){
                clearInterval(timer);
            }
        },1000);

    }

    waitVideo();

    document.addEventListener('yt-navigate-finish', () => {
        setTimeout(() => {
            reset();
        }, 1000);

    });

    setInterval(() => {
        const cap=document.querySelector('.ytp-caption-window-container');
        if(!cap){
            reset();
        }
    }, 3000);

})();
